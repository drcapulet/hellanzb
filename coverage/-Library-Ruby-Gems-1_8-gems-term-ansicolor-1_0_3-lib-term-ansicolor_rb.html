173     ATTRIBUTES.each do |c, v|
 
174       eval %Q{
 
175           def #{c}(string = nil)
 
176             result = ''
 
177             result << "\e[#{v}m" if Term::ANSIColor.coloring?
 
178             if block_given?
 
179               result << yield
 
180             elsif string
 
181               result << string
 
182             elsif respond_to?(:to_str)

184             else

186             end

188             result
 
189           end
 
190       }

193     # Regular expression that is used to scan for ANSI-sequences while
 
194     # uncoloring strings.
 
195     COLORED_REGEXP = /\e\[([34][0-7]|[0-9])m/
 
196 
197     # Returns an uncolored version of the string, that is all
 
198     # ANSI-sequences are stripped from the string.
 
199     def uncolored(string = nil) # :yields:
 
200       if block_given?
 
201         yield.gsub(COLORED_REGEXP, '')
 
202       elsif string
 
203         string.gsub(COLORED_REGEXP, '')
 
204       elsif respond_to?(:to_str)
 
205         gsub(COLORED_REGEXP, '')
 
206       else
 
207         ''
 
208       end
 
209     end
